            \begin{block}{\vspace*{0.5cm}\centering Compact matrix convex sets\vspace*{0.5cm}}\vspace*{0.25cm}
			A \textbf{compact matrix convex set over $\mathbf{V}$} is a disjoint union $K=\qty{K_n}_{n=1}^{\infty}$ of compact subsets $K_n\subset\mat{n}\qty(V)$ (and $\mat{n}\qty(V)$ is the set of $n\times n$ matrices with entries in $V$) which is closed under \textbf{matrix convex combinations} in $K$, namely:
\[
	v=\sum\limits_{i=1}^k\gamma_i^*v_i\gamma_i\in K_n
\]
for all $v_i\in K_{n_i}$ and $\gamma_i\in\mat{n_i}{n}$ ($i=1,\dots,k$) such that $\sum_{i=1}^k\gamma_i^*\gamma_i=I_n$.           \vspace*{1cm}

            \begin{block}{\vspace*{0.5cm}\centering Matrix extreme points\vspace*{0.5cm}}\vspace*{0.3cm}
            A point $v$ in a matrix convex set $K$ is called \textbf{matrix extreme} if it cannot be written as a matrix convex combination from lower levels in a non-trivial way. A point $v$ is called \textbf{absolute matrix extreme} if it cannot be written as any matrix convex combination in a non-trivial way. \par \vspace*{0.3cm}
            In (\cite[Theorem 4.3]{WebWink}), it was shown that the matrix extreme points of a compact matrix convex set generate it via matrix convex combinations.
            \end{block}

        \begin{block}{\vspace*{0.5cm}\centering Subhomogeneous matrix convex set \vspace*{0.5cm}}\vspace*{0.25cm}
        A compact matrix convex set $K$ over $V$ is called $\mathbf{n}$\textbf{-subhomogeneous} if there exists some $n\in\bN$ such that if $v\in\partial K$, then $v\in K_m$ for some $m\leq n$. \par\vspace*{0.5cm}
        \textbf{Proposition (}\cite[Proposition 3.5]{Kiri}\textbf{)}.\par
        Let $K$ be an $n$-subhomogeneous compact matrix convex set over a finite dimensional vector space $V$. Then, $K$ is the smallest matrix convex set over $V$ which has $K_n$ at level $n$.
        \end{block}

        \begin{block}{\vspace*{0.5cm}\centering Operator systems and CP maps \vspace*{0.5cm}}\vspace*{0.25cm}
                A subspace $\cS$ of a $C^*$-algebra is called an \textbf{operator system} if it is unital and self-adjoint. An element in $\cS$ is \textbf{positive} if it is positive as an element of $\cA$.\par\vspace{0.5cm} A linear mapping $\phi:\cS\to \cR$ between operator systems is $\mathbf{n}$\textbf{-positive} if $\qty(\phi\qty(s_{ij}))$ is positive in $\mat{n}\qty(\cR)$ whenever $\qty(s_{ij})$ is positive in $\mat{n}\qty(\cS)$. $\phi$ is \textbf{unital and completely positive (UCP)} if it is $n$-positive for all $n$, and $\phi(1_{\cS})=1_{\cR}$. When $\phi$ is UCP, invertible, and $\phi^{-1}$ is also UCP, we call it a \textbf{complete order isomorphism}.\par\vspace*{0.5cm} The disjoint union $\UCP{\cS}=\qty{\UCP{\cS}{\mat{n}}}_{n=1}^{\infty}$ of all UCP maps from $\cS$ to $\mat{n}$ is a prime example of a compact matrix convex set over $\cS^*$, which we call the \textbf{matrix state space} of $\cS$.  \par\vspace*{0.5cm}An operator system $\cS$ is called $\mathbf{n}$\textbf{-subhomogeneous} if $\UCP{\cS}$ is $n$-subhomogeneous.
            \end{block}
            \vspace*{1cm}
            \begin{highlightblock}{\vspace*{0.5cm}\centering What does $\UCP{\cS}$ teach us about $\cS$?\vspace*{0.5cm}}\vspace*{0.25cm}
\begin{itemize}
\item How does the normed structure of $\cS$ determine $\UCP{\cS}$ or its matrix extreme points?
\item How does the matrix convex structure of $\UCP{\cS}$ determine $\cS$ as an operator system?
\item Is the answer to the previous questions different if $\cS$ is $n$-subhomogeneous?
\item Is the answer to the previous questions different if $\cS$ is a $C^*$-algebra?
\item Can we understand general compact matrix convex set by interpolating the connection between operator systems and their matrix state spaces?
\end{itemize}
\end{highlightblock}

            \begin{block}{\vspace*{0.5cm}\centering General results\vspace*{0.5cm}}\vspace*{0.2cm}
           (\cite[Theorem B]{Farenick}) A matrix state $\phi\in\UCP{\cS}$ is a matrix extreme point if and only if it is pure.\par\vspace*{0.2cm}
            (\cite[Theorem 2.1]{Kiri})            A matrix state $\phi\in\UCP{\cS}$ is an absolute matrix extreme point if and only if it extends to an irreducible $*$-representation of $C^*\qty(\cS)$, such that the extension is the unique UCP extension of $\phi$ to $C^*\qty(\cS)$.\par \vspace*{0.2cm}
            (\cite[Proposition 4.1]{Kiri})            Two operator systems $\cS,\cR$ are completely order isomorphic if and only if $\UCP{\cS},\UCP{\cR}$ are matrix affine homeomorphic.\par \vspace*{0.2cm}
            (\cite[Corollary 2.6]{Kiri})            Let $K$ be a compact matrix convex set over $V$. A point $v\in \partial K\cap K_n$ is either an absolute matrix extreme point, or it is of the form $\gamma^*\tilde{v}\gamma$ for some $\tilde{v}\in \partial K\cap K_{n+1}$.
            \vspace*{0.2cm}\end{block}\vspace*{1cm}
            \begin{block}{\vspace*{0.5cm}\centering The subhomogeneous case\vspace*{0.5cm}}\vspace*{0.25cm}
            (\cite[Proposition 3.1]{Kiri}) Let $\cA$ be a unital $C^*$-algebra. Then $\cA$ is an $n$-subhomogeneous operator system if and only if its irreducible representations are of dimension at most $n$.\par\vspace*{0.2cm}
           Let $\cS,\cR$ be $n_{\cS}$-subhomogeneous and $n_{\cR}$-subhomogeneous operator systems, and set $n=\max\qty{n_{\cS},n_{\cR}}$.\par \vspace*{0.2cm}
           (\cite[Theorem 4.2]{Kiri}) $\cS$ and $\cR$ are $n$-order isomorphic if and only if they are completely order isomorphic.\par \vspace*{0.2cm}
           (\cite[Corollary 4.11]{Kiri}) If $\cS,\cR$ are generated by the $d$-tuples of unitaries $\bfs,\bfr$, the following are equivalent - (1) $s_i\mapsto r_i$ extends to an $n$-order isomorphism of $\cS$ and $\cR$. (2) $s_i\mapsto r_i$ extends to a complete order isomorphism of $\cS$ and $\cR$. (3) $s_i\mapsto r_i$ extends to a $*$-isomorphism of $C^*\qty(\cS)$ and $C^*\qty(\cR)$.\vspace*{0.2cm}
            \end{block}\vspace*{1cm}
